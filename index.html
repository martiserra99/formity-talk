Context

Problem

A way to define these forms and store them in a database

Formity

Package to build dynamic forms with JSON

Dependencies

Formity depends on React Hook Form and Mongu

React Hook Form

Manage React forms in a very simple way

Mongu

Create expressions written in JSON

            
              const { mongu } = require('mongu');

              const expr = {
                fullName: {
                  $concat: [{ $toLower: '$name' }, ' ', { $toLower: '$surname' }],
                },
                isAdult: { $gte: ['$age', 18] },
              };
              
              const vars = { name: 'Marti', surname: 'Serra', age: 24 };
              
              console.log(mongu(expr, vars)); // { fullName: "marti serra", isAdult: true }
            
          
            
              const { mongu } = require("mongu");

              const expr = { $add: ["$a", "$b"] };
              
              const vars = { a: 5, b: 10 };
              
              console.log(mongu(expr, vars)); // 15
            
          
            
              const { mongu } = require("mongu");

              const expr = {
                $cond: {
                  if: { $gte: ["$age", 18] },
                  then: "You are an adult",
                  else: "You are not an adult",
                },
              };
              
              const vars = { age: 18 };
              
              console.log(mongu(expr, vars)); // "You are an adult"
            
          
            
              const { mongu } = require("mongu");

              const expr = {
                $map: {
                  input: "$numbers",
                  as: "num",
                  in: { $multiply: ["$$num", 5] },
                },
              };
              
              const vars = { numbers: [1, 2, 3] };
              
              console.log(mongu(expr, vars)); // [5, 10, 15]
